% \newglossaryentry{__id__}{
%     name=,
%     description=,
%     first=,
%     plural=,
%     descriptionplural=,
%     firstplural=
% }
% __id__: id to use in the code with \gls{}, \Gls{}, \glspl{}, \Glspl{} and many others commands
%         (\glslink{}{} to write custom text but link it to an entry)
% name: main text used by the above commands (to access: \glsentrytext{__id__})
% description: printed in the glossary (to access: \glsentrydesc{__id__})
% first: to redefine the text printed at the first call (default: name content)
% plural: to redefine the plural form (default: name content with an s added at the end) (to access: \glsentryplural{__id__})
% descriptionplural: to define a plural form of the description, if needed
% firstplural: to redefine the text printed at the first call if its plural form (default: name content with an s added at the end)
% there is many other fields, see documentation if needed

\newglossaryentry{latex}{
    name=latex,
    description={Is a mark up language specially suited for scientific documents}
}

\newglossaryentry{seqAccesses}{
    name=sequential accesses,
    description={Are a group of memory accesses (from a data array, for example) that occur on conscutive memory addresses, in order through time},
    plural=sequential accesses
}

\newglossaryentry{randAccess}{
    name=``random'' access,
    description={Is a memory access that occurs on an arbitrary memory address (not necessarily conscutive in memory with previous and next accesses)},
    plural=``random'' accesses,
    sort=randomaccess
}

\newglossaryentry{randUpdt}{
    name=``random'' update,
    description={Is a \gls{randAccess} where the access is a read, modify and write on a given address. It is similar to a ``random'' \gls{reduction}},
    plural=``random'' updates,
    sort=randomupdate
}

\newglossaryentry{irrAccess}{
    name=irregular access,
    description={See \gls{randAccess}},
    plural=irregular accesses
}

\newglossaryentry{irrData}{
    name=irregular data,
    description={Refers to data that has low spatial and temporal locality, thus being more likely to generate \glspl{irrAccess}},
    plural=irregular data
}

\newglossaryentry{dFormat}{
    name=dense format,
    description={Is the uncompressed format where all zeros and non-zeros are stored, without any use of \gls{metadata}},
    plural=dense formats
}

\newglossaryentry{spFormat}{
    name=sparse format,
    description={Is a compression format for sparse matrices, which avoids storing zeros by using \gls{metadata}},
    plural=sparse formats
}

\newglossaryentry{metadata}{
    name=metadata,
    description={Is data that provides information about other data, such as index arrays of \glspl{spFormat}},
    plural=metadata
}

\newglossaryentry{tiling}{
    name=\emph{tiling},
    description={Is a method consisting in splitting data into several independent sub-regions to be processed separately},
    plural=\emph{tiling}
}

\newglossaryentry{tile}{
    name=\emph{tile},
    description={Refers to a sub-region of a \gls{tiling} process},
    plural=\emph{tiles}
}

\newglossaryentry{rank}{
    name=rank,
    description={Refers to a dimension of the kernel},
    plural=ranks
}

\newglossaryentry{crank}{
    name=common rank,
    description={Refers to a \gls{rank} that is common between two data-entries, when computing a kernel},
    plural=common ranks
}

\newglossaryentry{psum}{
    name=partial sum,
    description={Refers to the intermediary result before updating the output, when computing a matrix multiplication},
    plural=partial sums
}

\newglossaryentry{intersection}{
    name=\emph{intersection},
    description={Refers to the process of finding non-zero \glspl{psum}, when computing a matrix multiplication},
    plural=\emph{intersections}
}

\newglossaryentry{reduction}{
    name=\emph{reduction},
    description={Refers to the process of updating an entry},
    plural=\emph{reductions}
}

\newglossaryentry{fiber}{
    name=\emph{fiber},
    description={Is the generic term to describe a one-dimensional stream of data or \gls{metadata}. It corresponds to a row or column in the case of a two-dimensional matrix. A stream of indices is a \emph{fiber} of \gls{metadata}},
    plural=\emph{fibers}
}

\newglossaryentry{gather}{
    name=\emph{gather},
    description={Refers to reading data from memory. We specifically use this term to categorize data transfers that are irregular},
    plural=\emph{gather}
}

\newglossaryentry{scatter}{
    name=\emph{scatter},
    description={Refers to updating data in memory. We specifically use this term to categorize data transfers that are irregular},
    plural=\emph{scatter}
}

\newglossaryentry{band}{
    name=band,
    description={Refers to an area of a matrix that is concentrated around the diagonal},
    plural=bands
}

\newglossaryentry{banded}{
    name=banded,
    description={Refers to a matrix structure where the non-zero elements are concentrated around the diagonal},
    plural=banded
}

\newglossaryentry{PBanded}{
    name=`perfectly' banded,
    description={Refers to a matrix structure where all the non-zero elements are concentrated within a delimitated area around the diagonal (the \gls{band})},
    plural=`perfectly' banded,
    sort=perfectlybanded
}

\newglossaryentry{bandwidth}{
    name=\emph{bandwidth},
    description={With this typography, refers to the size of the \gls{band} of a matrix},
    plural=\emph{bandwidths}
}

\newglossaryentry{innefOp}{
    name=ineffectual operation,
    description={Is an operation that leads to a neutral result, $0$ for an addition or $1$ for a multiplication},
    plural=ineffectual operations
}

\newglossaryentry{hit}{
    name=hit,
    description={Is a cache event where a given address does match one of the valid cache entries},
    plural=hits
}

\newglossaryentry{miss}{
    name=miss,
    description={Is a cache event where a given address does not match any of the valid cache entries},
    plural=misses
}

\newglossaryentry{eviction}{
    name=eviction,
    description={Is a cache event where a valid cache entry is flushed and freed},
    plural=evictions
}

\newglossaryentry{redC}{
    name=\emph{reduction cache},
    description={Is a data-cache supporting \gls{reduction} operations},
    plural=\emph{reduction caches}
}
